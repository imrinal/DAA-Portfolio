Since Seetha wants to minimize her amount, she calls you to help her find the minimum cost that needs to be paid.

Solution


  #include <bits/stdc++.h>
  using namespace std;
  #define MAX_SIZE 1000005
  
  void SieveOfEratosthenes(vector<int>& primes){
      bool IsPrime[MAX_SIZE];
      memset(IsPrime, true, sizeof(IsPrime));
      for (int p = 2; p * p < MAX_SIZE; p++) {
          if (IsPrime[p]) {
              for (int i = p * p; i < MAX_SIZE; i += p)
                  IsPrime[i] = false;
          }
      }
      for (int p = 2; p < MAX_SIZE; p++)
          if (IsPrime[p])
              primes.push_back(p);
  }
  
  int main(){
      vector<int> primes;
      SieveOfEratosthenes(primes);
      int t;
      cin >> t;
      while(t--){
          int n;
          cin >> n;
          cout << primes[n-1] << endl;
      }
      return 0;
  }
